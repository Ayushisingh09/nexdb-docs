Getting Started

Installation

Rust (via Cargo)

cargo add nexdb

Adds NexDb to your Cargo.toml. Requires Rust 1.70+ and the tokio runtime.

Node.js (via npm)

npm install nexdb

Downloads a prebuilt binary for your platform, or builds from source if none is available.

Python (via pip)

pip install nexdb

Requires Rust toolchain for the initial build. The binary is compiled during pip install.

CLI Binary (standalone)

cargo install nexdb
nexdb repl mydb.nexdb

Quick Start

Rust

use nexdb::{NexDb, Document, NexDbResult};

#[tokio::main]
async fn main() -> NexDbResult<()> {
    let db = NexDb::open("hello.nexdb").await?;
    db.create_collection("greetings").await?;

    let doc = Document::from_json(r#"{"msg": "Hello, world!"}"#)?;
    db.insert("greetings", "g1", doc).await?;

    let saved = db.get("greetings", "g1").await?;
    println!("Got: {}", saved);

    db.flush().await?;
    Ok(())
}

JavaScript

const { NexDb } = require('nexdb');

async function main() {
    const db = new NexDb('hello.nexdb');
    await db.connect();
    await db.createCollection('greetings');
    const id = await db.insertAutoId('greetings', { msg: 'Hello, world!' });
    const doc = await db.get('greetings', id);
    console.log('Got:', doc);
    await db.close();
}
main().catch(console.error);

Python

from nexdb import NexDb

db = NexDb('hello.nexdb')
db.create_collection('greetings')
doc_id = db.insert_auto_id('greetings', {'msg': 'Hello, world!'})
doc = db.get('greetings', doc_id)
print('Got:', doc)
db.close()

CRUD Operations

// Create collection
db.create_collection("users").await?;

// Insert
let doc = Document::from_json(r#"{"name":"Alice","age":30}"#)?;
db.insert("users", "user-1", doc).await?;

// Auto-generated ID
let id = db.insert_auto_id("users", doc).await?;

// Read
let user = db.get("users", "user-1").await?;

// Update
let updated = Document::from_json(r#"{"name":"Alice","age":31}"#)?;
db.update("users", "user-1", updated).await?;

// Delete
db.delete("users", "user-1").await?;

// Count
let total = db.count("users").await?;

Indexes

NexDb supports field indexes for efficient equality and range queries.

// Create an index on the "age" field
db.create_index("users", "age_idx", "age").await?;

// Equality lookup
let results = db.find_eq("users", "age_idx",
    &serde_json::json!(30)).await?;

// Range lookup
let results = db.find_range("users", "age_idx",
    &serde_json::json!(20),
    &serde_json::json!(40)).await?;

WAL & Persistence

Every mutation is written to the Write-Ahead Log before the in-memory state is updated. On restart, the WAL is replayed to reconstruct the full database state.

// Flush WAL to disk explicitly
db.flush().await?;

// On next open, all data is recovered automatically:
let db = NexDb::open("myapp.nexdb").await?;
let count = db.count("users").await?;
// count == 3 (all previously inserted documents)
