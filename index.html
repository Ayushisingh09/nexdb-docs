Four Pillars. One Database.

Blazing Fast

In-memory first architecture with sub-millisecond latencies. Powered by Rust's zero-cost abstractions and BTreeMap-backed storage.

Rock Solid

Write-Ahead Logging (WAL) with fsync-guaranteed durability. Crash recovery via WAL replay. Deterministic memory — no GC pauses.

Schema-Less

Store anything as JSON documents. Dot-path field access. Auto-generated UUIDs or custom IDs. Collections self-describe.

Index-Powered

Pluggable field indexes over BTreeMap for equality and range queries. Foundation for future SQL parsing and query optimization.

Works Everywhere

use nexdb::{NexDb, Document};

#[tokio::main]
async fn main() -> nexdb::Result<()> {
    let db = NexDb::open("app.nexdb").await?;
    db.create_collection("users").await?;

    let doc = Document::from_json(r#"{
        "name": "Alice",
        "age": 30,
        "tags": ["admin", "dev"]
    }"#)?;

    db.insert("users", "u1", doc).await?;
    let user = db.get("users", "u1").await?;
    println!("{}", user);
    Ok(())
}
const { NexDb } = require('nexdb');

const db = new NexDb('app.nexdb');
await db.connect();

await db.createCollection('users');

const id = await db.insertAutoId('users', {
    name: 'Alice',
    age: 30,
    tags: ['admin', 'dev'],
});

const user = await db.get('users', id);
console.log(user);

await db.close();
from nexdb import NexDb

db = NexDb('app.nexdb')

db.create_collection('users')

user_id = db.insert_auto_id('users', {
    'name': 'Alice',
    'age': 30,
    'tags': ['admin', 'dev'],
})

user = db.get('users', user_id)
print(user)

db.close()
