// add to Cargo.toml
cargo add nexdb-core

// open database
let db = NexDb::open("app.nex")?;

// insert a document
let id = db.collection("users")
  .insert(json!({"name": "Alice", "age": 30}))?;

// query with filters
let results = db.collection("users")
  .query()
  .where_clause(r#"{"age": {"$gte": 21}}"#)?
  .sort("name", true)
  .limit(20)
  .run()?;
Features

Everything a modern database should be

Twelve capabilities. One cargo add. No daemon, no orchestrator, no ceremony.

Document Store

Schemaless JSON documents with dynamic queries, collections, and full CRUD — inspired by MongoDB.

WAL + Checkpoint

Append-only write-ahead log with snapshot-based checkpointing. Crash-safe by design, not by accident.

TCP Server

Built-in serve command. Connect over TCP with JSON-line protocol. Optional JWT auth.

Query Builder

Fluent .where(), .sort(), .limit() API with MongoDB-style operators.

B-tree Indexes

In-memory B-tree indexes on any field path. Automatic query optimization without configuration.

Auth & JWT

API key management and JWT-based authentication for secure TCP and cloud deployments.

Import / Export

Bulk JSON and CSV import and export from the CLI. Move data in seconds, not hours.

BSON Support

Optional binary document decoding via the bson feature flag. Small footprint by default.

Migration Engine

Dump, restore, copy between databases, export SQL schemas, and auto-import from any format.

Shell Completions

Generate auto-completions for bash, zsh, fish, PowerShell, and elvish in one command.

Benchmarked

Criterion-based benchmark suite for CRUD throughput and WAL write performance.

Clean Command

Remove single database files or bulk-wipe all NexDB databases in a directory.

Code

Three interfaces, one engine

Rust API, CLI, or TCP protocol — every operation goes through the same core.

use nexdb_core::NexDb; // open or create in one line let db = NexDb::open("mydb.nex")?; // collection CRUD let col = db.collection("users"); let id = col.insert(json!({"name": "Alice", 30}))?; let doc = col.get(&id)?; col.update(&id, json!({"name": "Alice", "age": 31}))?; col.delete(&id)?; // query with the builder let results = col .query() .where_clause(r#"{"age": {"$gte": 21, "$lt": 65}}"#)? .sort("name", true) .skip(0) .limit(20) .run()?; // create an index col.create_index("age")?;
# insert and query cargo nexdb insert mydb.nex users '{"name":"Bob","age":25}' cargo nexdb get mydb.nex users <id> cargo nexdb query mydb.nex users '{"age":{"$gte":21}}' # serve over TCP cargo nexdb serve --db mydb.nex --port 9480 # import and export cargo nexdb import mydb.nex users data.json cargo nexdb export mydb.nex users data.json cargo nexdb import-csv mydb.nex users data.csv # migrate between databases cargo nexdb migrate dump mydb.nex backup.json cargo nexdb migrate restore newdb.nex backup.json cargo nexdb migrate to-sql mydb.nex schema.sql # shell completions cargo nexdb completions bash > completions.sh # clean and checkpoint cargo nexdb checkpoint mydb.nex cargo nexdb clean mydb.nex
# start the server $ cargo nexdb serve --db mydb.nex --port 9480 # connect with nc $ nc localhost 9480 # every line is a JSON-RPC request: {"method": "insert", "params": { "collection": "users", "document": {"name": "Charlie", "age": 35} }} # response: {"ok": true, "id": "a1b2c3..."} {"method": "query", "params": { "collection": "users", "filter": {"age": {"$gte": 21}} }} # response: {"ok": true, "documents": [...]}
