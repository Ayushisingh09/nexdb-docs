// Add to Cargo.toml
cargo add nexdb-core

// Open database
let db = NexDb::open("app.nex")?;

// Insert a document
let id = db.collection("users")
  .insert(json!({"name": "Alice", "age": 30}))?;

// Query with filters
let results = db.collection("users")
  .query()
  .where_clause(r#"{"age": {"$gte": 21}}"#)?
  .sort("name", true)
  .limit(20)
  .run()?;
Features

Everything you expect from a modern database

NexDB packs 12+ features into a single Rust crate — no orchestrator, no daemon, no fuss.

📄

Document Store

Schemaless JSON documents with dynamic queries. Collections, indexes, and full CRUD out of the box.

⚡

WAL + Checkpoint

Append-only write-ahead log with snapshot checkpointing. Crash-safe by default.

📡

TCP Server

Built-in serve command. Connect over TCP with JSON-line protocol. Optional JWT auth.

🔍

Query Builder

Fluent .where(), .sort(), .limit() API with MongoDB-style operator support.

🗄️

B-tree Indexes

In-memory B-tree indexes on any document path. Automatic query optimization.

🔐

Auth & JWT

API key management and JWT authentication for secure TCP deployments.

📦

Import / Export

Bulk JSON and CSV import/export from the CLI. Pipeline data in seconds.

🧩

BSON Support

Optional binary document decoding via the bson feature flag.

🔄

Migration Engine

Dump, restore, copy, SQL export, and auto-import between databases.

⚙️

Shell Completions

Auto-completions for bash, zsh, fish, PowerShell, and elvish.

📊

Benchmarked

Criterion-based benchmark suite for CRUD and WAL throughput.

🧹

Clean Command

Remove database files individually or bulk-wipe all databases in a directory.

Code

See it in action

Choose your interface — Rust API, CLI, or TCP protocol. All backed by the same engine.

use nexdb_core::NexDb; // Open or create — one line let db = NexDb::open("mydb.nex")?; // Collection CRUD let col = db.collection("users"); let id = col.insert(json!({"name": "Alice", 30}))?; let doc = col.get(&id)?; col.update(&id, json!({"name": "Alice", "age": 31}))?; col.delete(&id)?; // Query with the builder let results = col .query() .where_clause(r#"{"age": {"$gte": 21, "$lt": 65}}"#)? .sort("name", true) .skip(0) .limit(20) .run()?; // Create an index col.create_index("age")?;
# Insert and query cargo nexdb insert mydb.nex users '{"name":"Bob","age":25}' cargo nexdb get mydb.nex users <id> cargo nexdb query mydb.nex users '{"age":{"$gte":21}}' # Serve over TCP cargo nexdb serve --db mydb.nex --port 9480 # Import / Export cargo nexdb import mydb.nex users data.json cargo nexdb export mydb.nex users data.json cargo nexdb import-csv mydb.nex users data.csv # Migrate between databases cargo nexdb migrate dump mydb.nex backup.json cargo nexdb migrate restore newdb.nex backup.json cargo nexdb migrate to-sql mydb.nex schema.sql # Shell completions cargo nexdb completions bash > completions.sh # Clean & checkpoint cargo nexdb checkpoint mydb.nex cargo nexdb clean mydb.nex
# Start the server $ cargo nexdb serve --db mydb.nex --port 9480 # Connect with nc $ nc localhost 9480 # Every line is a JSON-RPC request: {"method": "insert", "params": { "collection": "users", "document": {"name": "Charlie", "age": 35} }} # Response: {"ok": true, "id": "a1b2c3..."} {"method": "query", "params": { "collection": "users", "filter": {"age": {"$gte": 21}} }} # Response: {"ok": true, "documents": [...]}
